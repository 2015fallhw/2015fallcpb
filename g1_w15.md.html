2015 計算機程式 分組作業二

週五 25 十二月 2015 

#輸出
print('hello world')

#重複輸出
for i in range(4):
    print('hello world')

#添加句中間/句尾的記號
print('hello','world',sep="--",end='!')

#輸出範圍內隨機整數,int為integer(整數)的縮寫
import random
print(random.randint(0,99))

#利用while作條件,break來中斷,continue來迴圈
i = 1
j = 1
while i <= 9:
    if i == 8:
        break

    while j <= 9:
        if j == 4:
            j += 1
            continue

        print(i * j, end = " ")
        j += 1

    i += 1
    j = 1
    print()

#互動
x=input("輸入一個字")
print("你輸入了"+x)
