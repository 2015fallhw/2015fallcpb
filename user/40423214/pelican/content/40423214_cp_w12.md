#coding: utf-8
# 猜數字遊戲
import random
   
標準答案 = random.randint(1, 100)
你猜的數字 = int(input("請輸入您所猜的整數:"))
猜測次數 = 1
while 標準答案 != 你猜的數字:
    if 標準答案 < 你猜的數字:
        print("傻傻的，太大了，繼續猜")
    else:
        print("齁，太小，再猜")
    你猜的數字 = int(input("請輸入您所猜的整數:"))
    猜測次數 += 1
   
print("恭喜 你過github了", 猜測次數, "次")
